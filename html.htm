    <script>
        function placeBet() {
            const userNumber = document.getElementById('bet-number').value;
            const randomNumber = Math.floor(Math.random() * 10) + 1;

            if (userNumber == randomNumber) {
                document.getElementById('result').innerText = 'Parabéns! Você ganhou!';
            } else {
                document.getElementById('result').innerText = 'Que pena! Você perdeu. O número era ' + randomNumber;
            }
        }
